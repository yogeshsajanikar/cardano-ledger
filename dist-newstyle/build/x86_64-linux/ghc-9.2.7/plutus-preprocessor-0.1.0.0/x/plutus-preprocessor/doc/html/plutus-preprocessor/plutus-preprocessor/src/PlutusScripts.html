sumsTo10Decl :: Q [Dec]
sumsTo10Decl :: Q [Dec]
sumsTo10Decl =
  [d|
    sumsTo10' :: P.BuiltinData -> P.BuiltinData -> P.BuiltinData -> ()
    sumsTo10' d1 d2 _d3 =
      let n = P.unsafeDataAsI d1
          m = P.unsafeDataAsI d2
       in if (m P.+ n) P.== 10 then () else (P.error ())
    |]
